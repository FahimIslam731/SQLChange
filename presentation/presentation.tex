% 4. PARSING & ER GRAPH CONSTRUCTION
% ==========================================================================

\begin{frame}{Parsing \& ER Graph Construction}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Two Parsing Paths}\\[0.6em]
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.5em]{boxgray}
        \centering{\footnotesize\bfseries\color{aggieblue} Path 1: DDL Provided}\\[0.3em]
        \usebeamerfont{footline}\color{coolgray}
        \begin{flushleft}
        \textbf{sqlglot} parses DDL to extract tables, column types, primary/foreign keys, and constraints directly
        \end{flushleft}
      \end{beamercolorbox}
      \vspace{0.4em}
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.5em]{boxgray}
        \centering{\footnotesize\bfseries\color{aggieblue} Path 2: No DDL (Inference)}\\[0.3em]
        \usebeamerfont{footline}\color{coolgray}
        \begin{flushleft}
        Fast regex extracts table/column names from the query itself; LLM fallback infers types and relationships for ambiguous cases
        \end{flushleft}
      \end{beamercolorbox}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} ER Graph Construction}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item \textbf{Table importance hierarchy:}
          \begin{itemize}
            \setlength{\itemsep}{0.2em}
            \item \textit{Root} --- never a join target (changes cascade everywhere)
            \item \textit{Intermediate} --- both source and target
            \item \textit{Leaf} --- only a target (safest to modify)
          \end{itemize}
        \item \textbf{Cross-table risk signals:} graph depth, join-WHERE overlap, dependency fan-out
        \item Feeds context to recommendation engine and risk labeler downstream
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}

% ==========================================================================
% 5. RECOMMENDATION ENGINE
% ==========================================================================

\begin{frame}{Recommendation Engine}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} LLM-Powered Optimization}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item LLM receives \textbf{full upstream context}: parsed schema, ER graph, prior execution evidence, and iteration labels
        \item Generates optimized SQL with action type (\texttt{optimize}, \texttt{rewrite}, \texttt{keep\_original}, \texttt{flag\_for\_review})
        \item Outputs confidence score (1--10) and natural-language rationale
        \item On subsequent iterations: receives prior labels and adjusts conservatively if risk $\geq 7$
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Three-Layer Validation}\\[0.6em]
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.5em]{boxgray}
        \usebeamerfont{footline}\color{coolgray}
        \begin{flushleft}
        \textbf{\color{aggieblue}1. Syntax check} --- sqlglot must successfully parse the output as a valid SELECT/UNION/INTERSECT/EXCEPT\\[0.4em]
        \textbf{\color{aggieblue}2. SQLite compatibility} --- flags unsupported patterns (DATE\_SUB, INTERVAL, IF(), ILIKE, RIGHT JOIN) and suggests SQLite equivalents\\[0.4em]
        \textbf{\color{aggieblue}3. Schema reference} --- every table and column in the recommended SQL must exist in the parsed schema (allows aliases and CTEs)
        \end{flushleft}
      \end{beamercolorbox}
    \end{column}

  \end{columns}
\end{frame}

\begin{frame}{Execution Harness \& Labeling}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Synthetic Database Testing}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item Deterministic in-memory SQLite databases seeded by hash
        \item \textbf{Three scales:} 50 / 500 / 5{,}000 rows per table
        \item Realistic join conditions ($\sim$75\% match rate)
        \item Runs original vs.\ optimized on identical data
        \item \textbf{Equivalence:} identical / narrower / broader / different / error
        \item \textbf{Performance:} median of 5 runs per scale, speedup ratio
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Reasoning Labels}\\[0.6em]
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.4em]{boxgray}
        \centering{\footnotesize\bfseries\color{aggieblue} Semantic (equivalence)}\\[0.2em]
        \usebeamerfont{footline}\color{coolgray}
        Deterministic rules classify from execution output; LLM refines only when confidence is low
      \end{beamercolorbox}
      \vspace{0.3em}
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.4em]{boxgray}
        \centering{\footnotesize\bfseries\color{aggieblue} Performance (1--10)}\\[0.2em]
        \usebeamerfont{footline}\color{coolgray}
        Multi-scale timing analysis; 8--10 = significant improvement, 1--2 = severe degradation
      \end{beamercolorbox}
      \vspace{0.3em}
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.4em]{boxgray}
        \centering{\footnotesize\bfseries\color{aggieblue} Risk (1--10)}\\[0.2em]
        \usebeamerfont{footline}\color{coolgray}

      \end{beamercolorbox}
    \end{column}

  \end{columns}
\end{frame}

% ==========================================================================
% 7. ITERATIVE REFINEMENT
% ==========================================================================

\begin{frame}{Iterative Refinement}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Feedback Loop}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item Pipeline loops back if quality thresholds are not met
        \item All three labels (semantic, performance, risk) are fed back to the LLM on the next iteration
        \item LLM adjusts its recommendation based on what went wrong
        \item Default limit: \textbf{3 iterations} (configurable)
      \end{itemize}

      \vspace{0.6em}
      \setbeamercolor{boxblue}{bg=aggieblue,fg=white}
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,leftskip=0.7em,rightskip=0.7em,sep=0.5em]{boxblue}
        \centering\small\Lato\bfseries\color{white}
        Labels are computed deterministically first, then refined by LLM only when uncertain
      \end{beamercolorbox}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Stop Conditions}\\[0.6em]
      \begin{beamercolorbox}[wd=\textwidth,rounded=true,shadow=false,sep=0.5em]{boxgray}
        \usebeamerfont{footline}\color{coolgray}
        \begin{flushleft}
        Pipeline stops when \textbf{any} of these hold:\\[0.4em]
        \textbf{\color{aggieblue}1.} Performance $\geq 8$ \textbf{and} risk $\leq 3$\\[0.2em]
        \textbf{\color{aggieblue}2.} LLM recommends \texttt{keep\_original}\\[0.2em]
        \textbf{\color{aggieblue}3.} Generated SQL fails validation\\[0.2em]
        \textbf{\color{aggieblue}4.} Max iterations reached
        \end{flushleft}
      \end{beamercolorbox}

      \vspace{0.5em}
      {\small\Raleway\bfseries\color{aggieblue} Why Deterministic Scoring?}\\[0.4em]
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item \small Reproducible --- same input always yields same labels
        \item \small Cost-efficient --- avoids unnecessary LLM calls
        \item \small Auditable --- scoring logic is inspectable, not a black box
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}

\begin{frame}{Conclusions \& Future Work}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} What Worked}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item End-to-end pipeline from raw SQL to verified optimization
        \item Execution-based verification catches silent result-set changes
        \item Hybrid deterministic + LLM reasoning is cost-efficient and auditable
        \item Iterative feedback loop improves quality across passes
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Limitations \& Future Work}\\[0.6em]
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item Constrained to SQLite dialect --- future: PostgreSQL, MySQL
        \item Local 7B model (Qwen 2.5) --- larger models or fine-tuning may improve recommendation quality
        \item Synthetic data only --- real database connections for production use
        \item Expand beyond SQL: schema migration, stored procedures, query plan analysis
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}
