\begin{frame}{Recommendation Engine}
  \begin{columns}[T]

    \begin{column}{0.52\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} LLM-Powered Optimization}\\[0.5em]
      \small
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item LLM receives \textbf{full upstream context}: parsed schema, ER graph, prior execution evidence, and iteration labels
        \item Generates optimized SQL with action type (\texttt{optimize}, \texttt{rewrite}, \texttt{keep\_original}, \texttt{flag\_for\_review})
        \item Outputs confidence score (1--10) and natural-language rationale
        \item On subsequent iterations: receives prior labels and adjusts conservatively if risk $\geq 7$
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Three-Layer Validation}\\[0.5em]
      \small
      \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item \textbf{Syntax check} --- sqlglot must parse the output as valid SQL (SELECT, UNION, etc.)
        \item \textbf{SQLite compatibility} --- flags unsupported patterns (DATE\_SUB, INTERVAL, IF(), ILIKE, RIGHT JOIN) and suggests equivalents
        \item \textbf{Schema reference} --- every table and column in the recommended SQL must exist in the parsed schema (allows aliases and CTEs)
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}

\begin{frame}{Execution Harness \& Labeling}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Synthetic Database Testing}\\[0.5em]
      \small
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item Deterministic in-memory SQLite databases seeded by hash
        \item \textbf{Three scales:} 50 / 500 / 5{,}000 rows per table
        \item Realistic join conditions ($\sim$75\% match rate)
        \item Runs original vs.\ optimized on identical data
        \item \textbf{Equivalence:} identical / narrower / broader / different / error
        \item \textbf{Performance:} median of 5 runs per scale, speedup ratio
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Reasoning Labels}\\[0.5em]
      \small
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item \textbf{Semantic} (equivalence) --- deterministic rules classify from execution output; LLM refines only when confidence is low
        \item \textbf{Performance} (1--10) --- multi-scale timing analysis; 8--10 = significant improvement, 1--2 = severe degradation
        \item \textbf{Risk} (1--10) --- table importance, join depth, cross-table WHERE conditions, row-count mismatches
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}

% 7. ITERATIVE REFINEMENT
% ==========================================================================

\begin{frame}{Iterative Refinement}
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} How the Loop Works}\\[0.4em]
      \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item The execution harness runs both queries on synthetic data and produces equivalence + timing results
        \item Three labelers score the result: \textbf{semantic} (did the output change?), \textbf{performance} (is it faster?), \textbf{risk} (is it safe?)
        \item Deterministic rules assign labels first; the LLM only refines when confidence is low
        \item All scores feed back to the LLM, which generates an improved query with that context
        \item Loops until thresholds met (max 3 passes)
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Stop Conditions}\\[0.5em]
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item Performance $\geq 8$ \textbf{and} risk $\leq 3$
        \item LLM recommends \texttt{keep\_original}
        \item Generated SQL fails validation
        \item Max iterations reached (default: 3)
      \end{itemize}

      \vspace{0.6em}
      {\small\Raleway\bfseries\color{aggieblue} Per-Iteration Context}\\[0.5em]
      \begin{itemize}
        \setlength{\itemsep}{0.4em}
        \item Each iteration passes the prior iteration's scores, labels, and rationale to the LLM
        \item If risk $\geq 7$, the LLM switches to a conservative optimization strategy
        \item If semantic label is ``narrower'' or ``different'', the LLM prioritizes preserving the original result set
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}
