    \begin{column}{0.55\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} LLM-Powered Optimization}\\[0.4em]
      \footnotesize
      \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item LLM receives \textbf{full upstream context}: schema summary, ER graph hierarchy, execution evidence, and prior iteration labels
        \item Outputs: optimized SQL, action type (\texttt{optimize} / \texttt{rewrite} / \texttt{keep\_original}), confidence (1--10), rationale
        \item On later iterations: sees why the last attempt scored poorly and adjusts (conservative if risk $\geq 7$)
      \end{itemize}

      \vspace{0.3em}
      {\small\Raleway\bfseries\color{aggieblue} Three-Layer Validation}\\[0.4em]
      \footnotesize
      \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item \textbf{Syntax} --- sqlglot must parse output as valid SQL
        \item \textbf{SQLite compat} --- flags DATE\_SUB, INTERVAL, IF(), ILIKE, RIGHT JOIN; suggests equivalents
        \item \textbf{Schema ref} --- every table/column must exist in schema; validation fails $\to$ confidence downgraded
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}

    \begin{column}{0.55\textwidth}
      {\small\Raleway\bfseries\color{aggieblue} Synthetic Database}\\[0.4em]

      \begin{itemize}
        \setlength{\itemsep}{0.3em}
        \item Deterministic: seeded by query hash, reproducible across runs
        \item 75\% join match rate with realistic boundary values for WHERE clauses
        \item SQL normalized for SQLite (DATE\_SUB $\to$ strftime, true $\to$ 1)
      \end{itemize}

      \vspace{0.3em}
      {\small\Raleway\bfseries\color{aggieblue} Reasoning Labels}\\[0.4em]
      \footnotesize
      \begin{itemize}
        \setlength{\itemsep}{0.3em}
        \item \textbf{Semantic} --- deterministic classification from execution output (identical / narrower / broader / different); LLM refines only when confidence is low
        \item \textbf{Performance} (1--10) --- multi-scale timing; 8--10 = major speedup, 1--2 = degradation
        \item \textbf{Risk} (1--10) --- table importance hierarchy, join depth, cross-table WHERE, row-count mismatches
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}
